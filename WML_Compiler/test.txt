        <tr style="border: 1px solid #dddddd; text-align: left; padding: 8px;">
            <th><center>{{{title}}}</center></th>
        </tr>
        {{row|{{{buttons}}}|{{{filter}}} }} 
    </table>
:}

                    <td style="border: 1px solid #dddddd; text-align: left; padding: 8px;">
                        {{ {{car|{{{list}}} }}|toString}}
                    </td>
                </tr> |
            }}
            {{row|{{cdr|{{{list}}} }}|{{{filter}}} }}|
